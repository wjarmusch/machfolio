{{/* Reads the committed local snapshot; refresh via ./sync-data.command. */}} {{- $csv := resources.Get "data/entries.csv" -}} {{- if not $csv -}} {{- errorf "MachFolio: local data file assets/data/entries.csv not found. Run ./sync-data.command to fetch it from the sheet, then commit it." -}} {{- end -}} {{- $rows := $csv | transform.Unmarshal -}} {{- $headers := index $rows 0 -}} {{- $entries := slice -}} {{- range $i, $row := $rows -}} {{- if gt $i 0 -}} {{- $entry := dict -}} {{- range $j, $header := $headers -}} {{- $entry = merge $entry (dict $header (index $row $j)) -}} {{- end -}} {{- $entries = append $entry $entries -}} {{- end -}} {{- end -}} {{- return $entries -}}
